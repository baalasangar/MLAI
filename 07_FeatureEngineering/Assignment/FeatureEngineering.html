def checkoutliers(train_data):
  Q1 = train_data.quantile(0.25)
  Q3 = train_data.quantile(0.75)
  IQR = Q3 - Q1 
  outlier_data = pd.DataFrame(((train_data < (Q1 - 1.5 * IQR)) | (train_data > (Q3 + 1.5 * IQR))).sum(),columns=["outlier_count"])
  outlier_data.reset_index(inplace=True)
  outlier_data.rename(columns={"index":"feature"},inplace=True)
  outlier_data = outlier_data.sort_values(by="outlier_count")
  outlier_data =  outlier_data[outlier_data['outlier_count']>=1]
  return outlier_data

    • Using 3 oart liberaries for evaulation ( like MLxtend.paired_ttest_5x2cv() )
  • Second approach is used for the validation
